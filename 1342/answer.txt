class Solution {
    public int numberOfSteps(int num) {
        int count = 0;
        for (int i = num; num >= 0; i--) {
            if (num == 0) {
                break;
            }
            if (num % 2 == 0 && num != 0) {
                num = num / 2;
            } else {
                num = num - 1;
            }
            count++;
            if (num == 0) {
                break;
            }
        }
        return count;
    }
}
